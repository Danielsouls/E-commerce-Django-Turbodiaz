def acceso_login(request):
    if request.user.is_authenticated:
        return HttpResponseRedirect('/')

    form = Formulario_Login(request.POST or None)
    if request.method == 'POST':
        if form.is_valid():
            correo = form.cleaned_data['correo']
            password = form.cleaned_data['password']

            # Buscar usuario por correo y obtener su username
            try:
                user_obj = User.objects.get(email=correo)
                user = authenticate(request, username=user_obj.username, password=password)
            except User.DoesNotExist:
                user = None

            if user is not None:
                login(request, user)

                # Redirigir según el tipo de usuario
                if user.is_staff:
                    return HttpResponseRedirect('/admin/')  # Redirige al panel de admin
                else:
                    return HttpResponseRedirect('/')  # Redirige a la página principal

            else:
                messages.warning(request, 'Los datos ingresados no son correctos. Intenta nuevamente.')
                return HttpResponseRedirect('/acceso/login')

    return render(request, 'acceso/login.html', {'form': form})
